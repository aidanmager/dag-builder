USE ROLE SYSADMIN;

---

CREATE ROLE IF NOT EXISTS {{SF_ROLE}};

---

CREATE DATABASE IF NOT EXISTS {{DATABASE}};
GRANT USAGE ON DATABASE {{DATABASE}} TO ROLE {{SF_ROLE}};

---

USE DATABASE {{DATABASE}};

---

CREATE SCHEMA IF NOT EXISTS {{INGEST_SCHEMA}};
GRANT USAGE ON SCHEMA {{INGEST_SCHEMA}} TO ROLE {{SF_ROLE}};
GRANT SELECT,UPDATE,DELETE ON FUTURE TABLES
    IN SCHEMA {{INGEST_SCHEMA}} TO ROLE {{SF_ROLE}};

---

CREATE SCHEMA IF NOT EXISTS {{CLEANSE_SCHEMA}};
GRANT USAGE ON SCHEMA {{CLEANSE_SCHEMA}} TO ROLE {{SF_ROLE}};
GRANT SELECT,UPDATE,DELETE ON FUTURE TABLES 
    IN SCHEMA {{CLEANSE_SCHEMA}} TO ROLE {{SF_ROLE}};

---

CREATE SCHEMA IF NOT EXISTS {{READ_SCHEMA}};
GRANT USAGE ON SCHEMA {{READ_SCHEMA}} TO ROLE {{SF_ROLE}};
GRANT SELECT ON FUTURE VIEWS IN SCHEMA {{READ_SCHEMA}} TO ROLE {{SF_ROLE}};

---

CREATE OR REPLACE FILE FORMAT {{INGEST_SCHEMA}}.{{JSON_FORMAT}}
    TYPE = JSON
    STRIP_OUTER_ARRAY = TRUE
;
GRANT USAGE ON FILE FORMAT {{INGEST_SCHEMA}}.{{JSON_FORMAT}} TO ROLE {{SF_ROLE}};

---

CREATE OR REPLACE STAGE {{INGEST_SCHEMA}}.{{STAGE_NAME}}
    FILE_FORMAT=(FORMAT_NAME='{{DATABASE}}.{{INGEST_SCHEMA}}.{{JSON_FORMAT}}')
;
GRANT READ,WRITE ON STAGE {{INGEST_SCHEMA}}.{{STAGE_NAME}} TO ROLE {{SF_ROLE}};
